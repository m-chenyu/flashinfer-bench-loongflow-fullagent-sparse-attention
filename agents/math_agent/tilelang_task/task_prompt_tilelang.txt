import torch
import torch.nn as nn
import tilelang as T
from tilelang import tvm
# Define the TileLang kernel
@T.prim_func
def elementwise_add_kernel(
    a: T.Buffer((1, 128), "float32"),
    b: T.Buffer((1, 128), "float32"),
    out: T.Buffer((1, 128), "float32"),
    alpha: T.float32,
):
    # Specify the grid: divide the 128 elements into tiles (e.g., tile_size=128 for simplicity)
    with T.grid(1):
        with T.block(128):
            # Define the loop for the block
            for i in T.parallel(128):
                # Core mathematical logic implemented manually
                out[0, i] = alpha * a[0, i] + b[0, i]

# JIT Compile the kernel for PyTorch integration
# We specify the input shapes and types for tracing
input_a = torch.randn(1, 128).cuda()
input_b = torch.randn(1, 128).cuda()
output_out = torch.empty_like(input_a)

kernel_jit = T.jit.trace(
    elementwise_add_kernel,
    inputs=[input_a, input_b, output_out, 2.0],
    out_idx=[2] # Index of the output buffer in the prim_func arguments
)

class ModelNew(nn.Module):
    def __init__(self, alpha: float) -> None:
        super().__init__()
        self.alpha = alpha

    def forward(self, a, b):
        # TileLang JIT kernels are typically called with (input1, input2, ..., output, scalar_params)
        # Depending on the specific TileLang runtime version, the output can be pre-allocated
        out = torch.empty_like(a)
        kernel_jit(a, b, out, self.alpha)
        return out
